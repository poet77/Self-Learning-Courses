  z^5 + (-1 - 1901*x[a, b]/720)*z^4 + (2774*x[a, b]/720)*
    z^3 + (-2616*x[a, b]/720)*z^2 + (1274*x[a, b]/720)*
    z + (-x[a, b]*251/720);

y[a_, b_] := NSolve[f[z, a, b] == 0, z];
